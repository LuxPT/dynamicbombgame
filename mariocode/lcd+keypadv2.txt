#include <Keypad.h>
#include <LiquidCrystal.h>
//#include <Tone.h>


LiquidCrystal lcd(8,9,10,11,12,13);


char password[] ; 
int passe;
int posic = 0;
int tentativas=0;
char key;

const byte filas = 4;
const byte colunas = 4;
char tecla[filas][colunas] = {
{'1','2','3','A'},
{'4','5','6','B'},
{'7','8','9','C'},
{'*','0','#','D'}
};


byte pinFilas[filas] = { 3,4,5,6 };
byte pinColunas[colunas] = { 8,9,10,11 };
Keypad keypad = Keypad( makeKeymap(key), pinFilas, pinColunas, filas, colunas );

#define Redled A0	//incorreto
#define Greenled A1	//correto
#define led1 A2
#define led2 A3
#define led3 A4
#define piezo A5

String input_password = ""; //Stores the password typed by the user
//Armazena a senha digitada pelo usuário
int tentativas = 0; //Stores the number of attempts to authenticate 
//Armazena o número de tentativas de se autenticar 
bool locked = false; //When TRUE, the authentication is going to be blocked




void setup()
{
  lcd.begin(16,2);
  lcd.clear();
  Serial.begin(9600);


  pinMode(Redled, OUTPUT);
  pinMode(Greenled, OUTPUT);

  pinMode(led1, OUTPUT);
  pinMode(led2 ,OUTPUT);
  pinMode(led3,OUTPUT);
  pinMode(piezo, OUTPUT);

 noTone(piezo);                        //piezo calado
 digitalWrite(Redled, LOW);         //todos os leds apagados
 digitalWrite(Greenled,LOW);
 digitalWrite(led1,0);
 digitalWrite(led2,0);
 digitalWrite(led3,0);

passe=random(6);      

  lcd.setCursor(6,0);
  lcd.print("Arming the bomb:"); 
  delay(3000);
  lcd.clear();
  lcd.setCursor(6,0);
  Serial.println("Time is ticking");
  delay(3000);
  lcd.clear();
  lcd.setCursor(6,0);
  //lcd.print(time);
  

  lcd.setCursor(3,1);
  lcd.print("Enter the password: ");




switch(passe)
{
  case 0:			          //os leds apagados correspondem à pass F5678
  password = "F5678";
  digitalWrite(led1,0);
  digitalWrite(led2,0);
  digitalWrite(led3,0);
  break;
  

  case 1:				  //os leds1 e 2 apagados e o 3 aceso corresponde à pass 50D23
  password = "50D23"; 
  digitalWrite(led1,0);
  digitalWrite(led2,0);
  digitalWrite(led3,1);
   break;


case 2:				         //os leds leds2 e 3 acesos e o 3 apagado correspondem à pass 93524
  password = "93524";
  digitalWrite(led1,0);
  digitalWrite(led2,1);
  digitalWrite(led3,1);
  break;

case 3:				        //os leds acessos correspondem à pass 314A0
 password = "314A0";
 digitalWrite(led1,1);
 digitalWrite(led2,1);
 digitalWrite(led3,1);
 break;

case 4:				//os leds1 e 3 apagados  e 2 aceso corresponde à pass 0B2CD
 password = "0B2CD";
 digitalWrite(led1,0);
 digitalWrite(led2,1);
 digitalWrite(led3,0);
 break;


case 5:				//os leds2 e 3 apagados e o 1 aceso correspondem à pass 2B699
  password = "2B699";
  digitalWrite(led1,1);
  digitalWrite(led2,0);
  digitalWrite(led3,0);
  break;

}


}

void loop()
{
    lcd.clear();
  	lcd.print("Senha:");
  	do {
    
      key = keypad.getKey();
      if((key != NO_KEY) && (key != '*') && (key != '#'))
      {
      	lcd.print(key);
      	input_password += key;
      }
    
    
    } while((key == NO_KEY) || ( (key != '*') && (key != '#') )); 
  
    if(input_password == password)
    { 
      posic ++;
      digitalWrite(Greenled, HIGH);		//o led verde acende (o vermelho já estava apagado i guess)
      digitalWrite(Redled, LOW);					// os leds piscam numa sequencia toda bonita tipo cascata (talvez dê para simplificar esse codigo seguinte mas tenho preguiça) e depois apagam
      digitalWrite(led1,0);
      digitalWrite(led2,0);
      digitalWrite(led3,0);
      delay(500);

      digitalWrite(led1,1);
      digitalWrite(led2,0);
      digitalWrite(led3,0);
       delay(500);

      digitalWrite(led1,0);
      digitalWrite(led2,1);
      digitalWrite(led3,0);
     delay(500);

      digitalWrite(led1,0);
      digitalWrite(led2,0);
      digitalWrite(led3,1);
     delay(500);

     digitalWrite(led1,0);
     digitalWrite(led2,1);
     digitalWrite(led3,0);
     delay(500);
 
     digitalWrite(led1,1);
     digitalWrite(led2,0);
     digitalWrite(led3,0);
      delay(500);

     digitalWrite(led1,0);
     digitalWrite(led2,0);
     digitalWrite(led3,0);

     delay(500);		//o piezo apita a seguir 2 vezes
     tone(piezo, 980, 500);
    delay(500);
    tone(piezo, 980, 500);

lcd.print("The bomb has been defused");

    }
    else
    {	tone(piezo, 280, 500);
      lcd.clear();
      lcd.print("ACESSO NEGADO!");
      Serial.println("\nAcesso negado!");
     tone(piezo, 280, 500);		//o piezo apita o som de erro 2 vezes
     delay(500);
     tone(piezo, 280, 500);
      tentativas++;
    }
  
    if(tentativas >= 4)
    {
      
      lcd.clear();
      lcd.print("tempo-30 ");
     
    }
    
    input_password = "";
    delay(5000);
  }
}
